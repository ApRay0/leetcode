class Solution {
public:
    ListNode* rotateRight(ListNode* head, int k) {
        if (head==NULL) return head;
        ListNode* p,*q;
        p=q=head;
        int i=1;
        while(p->next){
            p=p->next;
            i++;
        }
        p->next=head;
        k=k%i;
        k=i-k;   
        while(k){
            k--;
            q=q->next;
            p=p->next;
        }
        p->next=NULL;
        return q;
    }
};
