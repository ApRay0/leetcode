class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int n=s.size(),res=0;
        map<char,int> map;
        for (int i=0,j=0;j<n;j++){
            if(map.find(s.at(j))!=map.end()){
                i=max(i,map[s.at(j)]);
            }
               res=max(res,j-i+1);
               map[s[j]]=j+1;
        }
        return res;
    }
};
